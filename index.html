using System;
using System.Drawing;
using System.Windows.Forms;

namespace WinFormsApp1 // DİKKAT: Burayı kendi proje adınla değiştir!
{
    public partial class Form1 : Form
    {
        int sayac = 30;

        public Form1()
        {
            InitializeComponent();
        }

        private void Form1_Load(object sender, EventArgs e)
        {
            SorulariAktifEt(false);
            lblSure.Text = "30";
            // Başlangıçta sonuç etiketlerini gizle veya boşalt
            lblDogruSayisi.Text = "0";
            lblYanlisSayisi.Text = "0";
            lblSonucBilgi.Visible = false;
        }

        private void btnBaslat_Click(object sender, EventArgs e)
        {
            // Giriş kontrolü
            if (string.IsNullOrEmpty(txtTC.Text) || string.IsNullOrEmpty(txtAdSoyad.Text))
            {
                MessageBox.Show("Lütfen TC No ve Ad Soyad alanlarını doldurunuz!", "Uyarı", MessageBoxButtons.OK, MessageBoxIcon.Warning);
                return;
            }

            if (txtTC.Text.Length != 11)
            {
                MessageBox.Show("T.C. No 11 haneden oluşmalıdır!", "Hata", MessageBoxButtons.OK, MessageBoxIcon.Error);
                return;
            }

            // Sınavı başlat
            SorulariAktifEt(true);
            btnBaslat.Enabled = false;
            txtTC.Enabled = false;
            txtAdSoyad.Enabled = false;

            sayac = 30;
            timer1.Interval = 1000;
            timer1.Start();
        }

        private void timer1_Tick(object sender, EventArgs e)
        {
            sayac--;
            lblSure.Text = sayac.ToString();

            if (sayac <= 0)
            {
                timer1.Stop();
                TestiBitir();
                MessageBox.Show("Süre Doldu!", "Bilgi", MessageBoxButtons.OK, MessageBoxIcon.Information);
            }
        }

        private void btnBitir_Click(object sender, EventArgs e)
        {
            timer1.Stop();
            TestiBitir();
        }

        private void SorulariAktifEt(bool durum)
        {
            groupBox1.Enabled = durum;
            groupBox2.Enabled = durum;
            groupBox3.Enabled = durum;
            groupBox4.Enabled = durum;
        }

        private void TestiBitir()
        {
            SorulariAktifEt(false);
            btnBitir.Enabled = false;

            int dogru = 0;
            int yanlis = 0;

            // --- SORU 1: Veri türü değildir? (Cevap: Checkbox - D Şıkkı) ---
            if (rdbSoru1D.Checked)
            {
                dogru++;
                rdbSoru1D.BackColor = Color.LightGreen;
            }
            else
            {
                if (rdbSoru1A.Checked || rdbSoru1B.Checked || rdbSoru1C.Checked) yanlis++;
                rdbSoru1D.BackColor = Color.LightGreen; // Doğruyu göster
            }

            // --- SORU 2: Timer milisaniye? (Cevap: Interval - A Şıkkı) ---
            if (rdbSoru2A.Checked)
            {
                dogru++;
                rdbSoru2A.BackColor = Color.LightGreen;
            }
            else
            {
                if (rdbSoru2B.Checked || rdbSoru2C.Checked || rdbSoru2D.Checked) yanlis++;
                rdbSoru2A.BackColor = Color.LightGreen;
            }

            // --- SORU 3: ForeColor özelliği? (Cevap: Yazı Rengi - B Şıkkı) ---
            if (rdbSoru3B.Checked)
            {
                dogru++;
                rdbSoru3B.BackColor = Color.LightGreen;
            }
            else
            {
                if (rdbSoru3A.Checked || rdbSoru3C.Checked || rdbSoru3D.Checked) yanlis++;
                rdbSoru3B.BackColor = Color.LightGreen;
            }

            // --- SORU 4: c+=5 anlamı? (Cevap: 5 artırır - C Şıkkı) ---
            if (rdbSoru4C.Checked)
            {
                dogru++;
                rdbSoru4C.BackColor = Color.LightGreen;
            }
            else
            {
                if (rdbSoru4A.Checked || rdbSoru4B.Checked || rdbSoru4D.Checked) yanlis++;
                rdbSoru4C.BackColor = Color.LightGreen;
            }

            // Sonuçları Yazdır
            lblDogruSayisi.Text = dogru.ToString();
            lblYanlisSayisi.Text = yanlis.ToString();

            lblSonucBilgi.Visible = true;
            lblSonucBilgi.Text = $"Tc No: {txtTC.Text}\nAd Soyad: {txtAdSoyad.Text}";
        }
    }
}

Form1.cs 
using System;
using System.Drawing;
using System.Windows.Forms;

namespace WinFormsApp1
{
    public partial class Form1 : Form
    {
        public Form1()
        {
            InitializeComponent();
        }

        // Form Yüklendiğinde Başlangıç Ayarları
        private void Form1_Load(object sender, EventArgs e)
        {
            // Timer'ı başlatıyoruz (Animasyon için gerekli)
            timer1.Interval = 50; // Hız ayarı (düşük sayı = hızlı hareket)
            timer1.Start();

            // Başlangıçta radyo butonu seçimine göre görünürlükleri ayarla
            GorumurlukleriAyarla();
        }

        // Radyo Butonları Değiştiğinde Çalışacak Metot
        // Designer kısmında tüm radio button'ların CheckedChanged olayına bu metodu bağladım.
        private void SecimDegisti(object sender, EventArgs e)
        {
            GorumurlukleriAyarla();
        }

        private void GorumurlukleriAyarla()
        {
            // b) Seçili olan işlemin nesnesi ekranda görünecek, diğerleri görünmeyecek.

            // 1. Durum: Kayan Yazı
            lblKayanYazi.Visible = rbKayanYazi.Checked;
            if (rbKayanYazi.Checked)
            {
                // Pozisyonu sıfırla
                lblKayanYazi.Left = 0;
            }

            // 2. Durum: Kare Hareket
            pnlKare.Visible = rbKareHareket.Checked;
            if (rbKareHareket.Checked)
            {
                // Pozisyonu sıfırla
                pnlKare.Left = 0;
            }

            // 3. Durum: Nesne Büyütme
            pnlBuyuyen.Visible = rbNesneBuyutme.Checked;
            if (rbNesneBuyutme.Checked)
            {
                // c) Boyutları 25x25 olarak sıfırla
                pnlBuyuyen.Size = new Size(25, 25);
                // Ortaya konumlandır (İsteğe bağlı, görsel güzellik için)
                pnlBuyuyen.Location = new Point((this.ClientSize.Width - 25) / 2, (this.ClientSize.Height - 25) / 2);
            }
        }

        // Timer Her Tetiklendiğinde (Animasyon Döngüsü)
        private void timer1_Tick(object sender, EventArgs e)
        {
            // a) ve d) Kayan Yazı İşlemleri
            if (rbKayanYazi.Checked)
            {
                lblKayanYazi.Left += 5; // Sağa doğru 5 piksel kaydır

                // f) Ekrandan çıkınca tekrar başla
                if (lblKayanYazi.Left > this.ClientSize.Width)
                {
                    lblKayanYazi.Left = -lblKayanYazi.Width;
                }
            }

            // a) ve e) Kare Hareket İşlemleri
            if (rbKareHareket.Checked)
            {
                pnlKare.Left += 5; // Sağa doğru 5 piksel kaydır

                // f) Ekrandan çıkınca tekrar başla
                if (pnlKare.Left > this.ClientSize.Width)
                {
                    pnlKare.Left = -pnlKare.Width;
                }
            }

            // a) ve c) Nesne Büyütme İşlemleri
            if (rbNesneBuyutme.Checked)
            {
                // Nesneyi büyüt
                pnlBuyuyen.Width += 2;
                pnlBuyuyen.Height += 2;

                // Konumunu merkeze göre tekrar ayarla ki ortadan büyüsün (İsteğe bağlı estetik ayar)
                pnlBuyuyen.Left -= 1;
                pnlBuyuyen.Top -= 1;

                // f) Ekrandan taşınca veya çok büyüyünce tekrar başla
                // Formun genişliğinden büyük olursa sıfırla
                if (pnlBuyuyen.Width > this.ClientSize.Width || pnlBuyuyen.Height > this.ClientSize.Height)
                {
                    pnlBuyuyen.Size = new Size(25, 25);
                    pnlBuyuyen.Location = new Point((this.ClientSize.Width - 25) / 2, (this.ClientSize.Height - 25) / 2);
                }
            }
        }
    }
